Prebuilt pipelines for common genomic analyses

Differential Expression Analysis

dea_pipeline <- function(se, design) {
  se |> 
    normalize(method = "TMM") |>
    model_glm(design = design) |>
    test_contrasts(
      contrast = c("condition", "treated", "control"),
      method = "DESeq2"
    ) |>
    add_annotations(org_db = "org.Hs.eg.db")
}

ChIP-seq Peak Analysis

chip_pipeline <- function(se) {
  se |>
    call_peaks(method = "MACS2") |>
    annotate_peaks(txdb = TxDb.Hsapiens.UCSC.hg38.knownGene) |>
    motif_analysis(species = "Hsapiens")
}

tcga_se <- importTCGA(

2. Quality Control

qc_report(tcga_se, 
          output = "tcga_qc.html",
          metrics = c("detection_rate", "library_complexity"))

3. Survival Analysis

surv_res <- analyze_survival(
  tcga_se,
  time_var = "OS.time",
  event_var = "OS",
  gene_list = c("BRCA1", "BRCA2")
)
