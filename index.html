mergeSE()

Data Integration

Merge multiple SummarizedExperiment objects

Syntax

mergeSE(..., 
       by.row = NULL,
       by.col = NULL,
       assay.merge = "union",
       metadata.resolve = "smart")

Parameters

...
SE objects to merge (minimum 2)
by.row
Row identifiers for merging (e.g., gene symbols)
by.col
Column identifiers for merging (e.g., sample IDs)
assay.merge
"union" or "intersection" strategy
metadata.resolve
Conflict resolution strategy ("smart", "first", "list")

Example

# Merge RNA-seq and proteomics data
merged_se <- mergeSE(
    rna_se, 
    prot_se,
    by.row = "gene_symbol",
    by.col = "patient_id",
    assay.merge = "union"
)
dim(merged_se)
# Output: [1] 25000   150

subsetSE()

Data Filtering

Intelligent subsetting of SE objects

subsetSE(se, 
        rows = NULL, 
        cols = NULL,
        assays = NULL,
        mode = "strict")
# Subset protein-coding genes and tumor samples
sub_se <- subsetSE(se,
                  rows = rowData(se)$gene_type == "protein_coding",
                  cols = colData(se)$group == "tumor",
                  assays = c("counts", "logcounts"))

find_markers()

Differential expression analysis

find_markers(se, 
            group_col,
            method = c("DESeq2", "edgeR", "limma"),
            design = NULL,
            contrast = NULL)
# DESeq2 analysis
de_res <- find_markers(se,
                      group_col = "condition",
                      method = "DESeq2",
                      contrast = c("treated", "control"))

# edgeR-QLF analysis
de_res <- find_markers(se,
                      method = "edgeR",
                      design = ~ batch + condition)

enrich_analysis()

Functional enrichment analysis

enrich_analysis(de_results, 
               org_db = "org.Hs.eg.db",
               ontology = "BP",
               p_cutoff = 0.05)
go_res <- enrich_analysis(de_res,
                         ontology = "MF",
                         p_cutoff = 0.01)
dotplot(go_res, showCategory=20)
