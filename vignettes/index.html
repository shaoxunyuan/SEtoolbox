ğŸš€ Quick Start

Basic Workflow

library(SEtoolbox)
library(SummarizedExperiment)

# Load example dataset
data(se)

# Quality control
se <- run_qc(se, plot=TRUE)

# Differential analysis
de_results <- find_markers(se, 
                          group_col = "condition",
                          method = "DESeq2")

# Visualize results
volcano_plot(de_results)
